}

activate etl

loop
   etl-> dbtemp: Конвертация данных
   activate dbtemp

   deactivate dbtemp
end

loop
   etl-> dbtemp: Сортировка данных
   activate dbtemp
   dbtemp--> etl: Поток данных
   deactivate dbtemp
end

loop
   etl-> dbtemp: Группировка данных
   activate dbtemp
   dbtemp--> etl: Поток данных
   deactivate dbtemp
end


== Споставление данных ==


   etl--> dbdst: поток данных
   deactivate etl

== ==
